		int[] arr = new int[n];

			arr[i] = i+1;
		}
		for(int rt = 1; rt < arr.length; rt++) {
			int sum = 0;
			for(int i = lt; i <= rt; i++) {
				sum += arr[i];
			}
			if(sum == n) { 
				answer++;
				lt++;
			}
			while( sum > n ) {
				sum -= arr[lt++]; 
				if(lt == n-1 ) break;
				if(sum == n ) {
					answer++;
				}
			}
		}
		return answer;
	}
}
