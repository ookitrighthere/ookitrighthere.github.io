	static int solution(int n , int[] arr) {
		ArrayList<Integer> answer = new ArrayList<>();
		answer.add(arr[0]);
		int max = arr[0];
		for(int i = 1; i < n; i++) {
			if(arr[i] > max ) {
				max = arr[i];
				answer.add(arr[i]);
			}
		}
		return answer.size();
	}
}
