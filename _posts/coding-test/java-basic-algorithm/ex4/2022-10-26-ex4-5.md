	static int solution(int n, int k, int[] arr) {
		int answer = -1;
		TreeSet<Integer> set = new TreeSet<>(Collections.reverseOrder());
		for(int i = 0; i < n; i++) {
			for(int j = i+1; j < n; j++) {
				for(int l = j+1; l < n; l++) {
					set.add(arr[i] + arr[j] + arr[l] );
				}
			}
		}
		int count = 0;
		for(int i  : set) {
			count++;
			if(count == k) {
				return i;
			}
		}
		return answer;
	}
}
