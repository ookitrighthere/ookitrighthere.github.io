	static int solution(String str) {
		Stack<Integer> stack = new Stack<>();
		for(char c : str.toCharArray()) {
			if(Character.isDigit(c)) {
				stack.push(c-48);
			}else {
				int rt = stack.pop();
				int lt = stack.pop();
				if(c == '+') stack.push(lt + rt);
				else if (c == '-') stack.push(lt - rt);
				else if (c == '*') stack.push(lt * rt);
				else if (c == '/') stack.push(lt / rt);
			}
		}
		answer=stack.get(0);
		return answer;
	}
}
