class Solution {
    private static int[] dx = {0, 1, 0, -1};
    private static int[] dy = {-1, 0, 1, 0};


    public int solution(int[][] maps) {
        int[][] visited = new int[maps.length][maps[0].length];

        Queue<State> q = new LinkedList<>();
        q.add(new State(0,0,1));
        visited[0][0]=1;

        int targetY = maps.length-1;
        int targetX = maps[targetY].length-1;
        while(!q.isEmpty()) {
            State state = q.poll();
            if(state.x == targetX && state.y == targetY) {
                return state.step;
            }

            for(int i = 0; i < dx.length; i++) {
                int x = state.x + dx[i];
                int y = state.y + dy[i];

                if(y < 0 || x < 0 || y >= maps.length || x >= maps[y].length) {
                    continue;
                }

                if(maps[y][x] != 1) {
                    continue;
                }

                if(visited[y][x] == 1 ) {
                    continue;
                }
                visited[y][x] = 1;
                q.add(new State(x, y, state.step + 1));

            }
        }

        return -1;
    }
}

class State {
    int x;
    int y;
    int step;

    public State(int x, int y, int step) {
        this.x = x;
        this.y = y;
        this.step = step;
    }
}
