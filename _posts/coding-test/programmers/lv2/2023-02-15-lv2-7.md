```java
class Solution {
    public int solution(int n) {
		int answer = 0;
        if (n == 1 || n ==2 ) return 1;
		int sum=0;
		int m = n/2+1;
		int[] arr=new int[m];
		for(int i = 0; i < m; i++) arr[i]=i+1;
		int lt=0;
		for(int rt=0; rt<m; rt++){
			sum+=arr[rt];
			if(sum == n) answer++;
			
			while(sum >= n){
				sum-=arr[lt++];
				if(sum == n) answer++; 
			}
		}
		return ++answer;
    }
}
```
